# 使用Python实现简单的AES加密解密示例 下面是一个使用Python的`cryptography`库实现AES加密解密的完整示例。这个示例包含了： 1. 生成随机密钥 2. 使用CBC模式进行加密 3. 使用PKCS7填充 4. 完整的加密和解密函数 ```python import os from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes from cryptography.hazmat.primitives import padding from cryptography.hazmat.backends import default_backend import base64 class AESEncryptor: def __init__(self, key=None): """ 初始化AES加密器 Args: key: 可选的密钥（16, 24或32字节）。如果为None，则生成随机密钥 """ if key is None: # 生成随机密钥（AES-256需要32字节） self.key = os.urandom(32) else: if len(key) not in [16, 24, 32]: raise ValueError("密钥长度必须为16, 24或32字节") self.key = key def encrypt(self, plaintext): """ 使用AES-CBC模式加密文本 Args: plaintext: 要加密的文本（字符串） Returns: tuple: (加密后的base64字符串, 初始化向量base64字符串) """ # 生成随机初始化向量（IV） iv = os.urandom(16) # 创建AES-CBC加密器 cipher = Cipher( algorithms.AES(self.key), modes.CBC(iv), backend=default_backend() ) # 创建加密器对象 encryptor = cipher.encryptor() # 对明文进行PKCS7填充 padder = padding.PKCS7(128).padder() padded_data = padder.update(plaintext.encode('utf-8')) + padder.finalize() # 加密数据 ciphertext = encryptor.update(padded_data) + encryptor.finalize() # 返回base64编码的密文和IV return ( base64.b64encode(ciphertext).decode('utf-8'), base64.b64encode(iv).decode('utf-8') ) def decrypt(self, ciphertext_b64, iv_b64): """ 解密AES-CBC加密的文本 Args: ciphertext_b64: base64编码的密文 iv_b64: base64编码的初始化向量 Returns: str: 解密后的原始文本 """ # 解码base64数据 ciphertext = base64.b64decode(ciphertext_b64) iv = base64.b64decode(iv_b64) # 创建AES-CBC解密器 cipher = Cipher( algorithms.AES(self.key), modes.CBC(iv), backend=default_backend() ) # 创建解密器对象 decryptor = cipher.decryptor() # 解密数据 padded_plaintext = decryptor.update(ciphertext) + decryptor.finalize() # 移除PKCS7填充 unpadder = padding.PKCS7(128).unpadder() plaintext = unpadder.update(padded_plaintext) + unpadder.finalize() return plaintext.decode('utf-8') def get_key_b64(self): """获取base64编码的密钥""" return base64.b64encode(self.key).decode('utf-8') @staticmethod def create_from_key_b64(key_b64): """从base64编码的密钥创建加密器""" key = base64.b64decode(key_b64) return AESEncryptor(key) def main(): """主函数：演示AES加密解密的完整流程""" print("=" * 50) print("AES加密解密演示") print("=" * 50) # 1. 创建AES加密器（自动生成密钥） print(" 1. 创建AES加密器...") encryptor = AESEncryptor() print(f"生成的密钥（base64）: {encryptor.get_key_b64()}") print(f"密钥长度: {len(encryptor.key)} 字节") # 2. 要加密的原文 original_text = "这是一段需要加密的敏感信息，包含中文和英文：Hello World!" print(f" 2. 原始文本: {original_text}") # 3. 加密 print(" 3. 加密中...") ciphertext_b64, iv_b64 = encryptor.encrypt(original_text) print(f"密文（base64）: {ciphertext_b64[:50]}...") print(f"初始化向量（base64）: {iv_b64}") # 4. 解密 print(" 4. 解密中...") decrypted_text = encryptor.decrypt(ciphertext_b64, iv_b64) print(f"解密后的文本: {decrypted_text}") # 5. 验证 print(" 5. 验证结果...") if original_text == decrypted_text: print("✓ 加密解密成功！原文与解密后文本一致。") else: print("✗ 加密解密失败！") # 6. 演示使用固定密钥 print(" " + "=" * 50) print("使用固定密钥的示例") print("=" * 50) # 使用固定密钥（AES-128，16字节） fixed_key = b'0123456789abcdef' # 16字节密钥 fixed_encryptor = AESEncryptor(fixed_key) test_text = "使用固定密钥加密的测试文本" print(f" 原始文本: {test_text}") # 加密 ciphertext_b64, iv_b64 = fixed_encryptor.encrypt(test_text) print(f"密文: {ciphertext_b64}") # 使用相同的密钥创建新的加密器进行解密 key_b64 = fixed_encryptor.get_key_b64() new_encryptor = AESEncryptor.create_from_key_b64(key_b64) decrypted_text = new_encryptor.decrypt(ciphertext_b64, iv_b64) print(f"解密后: {decrypted_text}") if __name__ == "__main__": # 安装依赖（如果未安装） try: from cryptography.hazmat.primitives.ciphers import Cipher except ImportError: print("正在安装所需依赖...") import subprocess import sys subprocess.check_call([sys.executable, "-m", "pip", "install", "cryptography"]) print("依赖安装完成，重新运行程序。") else: main() ``` ## 使用说明 ### 1. 安装依赖 首先需要安装`cryptography`库： ```bash pip install cryptography ``` ### 2. 基本用法 ```python # 创建加密器（自动生成密钥） encryptor = AESEncryptor() # 加密 ciphertext, iv = encryptor.encrypt("要加密的文本") # 解密 plaintext = encryptor.decrypt(ciphertext, iv) # 保存和加载密钥 key_b64 = encryptor.get_key_b64() # 保存这个密钥 new_encryptor = AESEncryptor.create_from_key_b64(key_b64) # 用保存的密钥创建新加密器 ``` ### 3. 重要注意事项 1. **密钥管理**：密钥必须安全存储，丢失密钥将无法解密数据 2. **初始化向量（IV）**：每次加密都应使用不同的随机IV，IV不需要保密但需要与密文一起存储 3. **密钥长度**： - 16字节 = AES-128 - 24字节 = AES-192 - 32字节 = AES-256 4. **安全建议**： - 在生产环境中使用密钥管理系统 - 定期轮换密钥 - 使用适当的密钥派生函数（如PBKDF2）从密码生成密钥 ## 运行示例 直接运行上面的代码即可看到完整的加密解密演示。程序会： 1. 自动生成AES密钥 2. 加密一段包含中文和英文的文本 3. 解密并验证结果 4. 演示使用固定密钥的场景 这个示例使用了AES-CBC模式，这是最常用的对称加密模式之一，适合大多数应用场景。
